## Installation
```bash
npx shadcn@latest add your-registry-url/iconic-heading
```

## Usage
```tsx
import { IconicHeading } from "@/components/Icon-ic-Heading"
import { TrendingUp } from "lucide-react"
```
```tsx
<IconicHeading
  lines={[
    {
      text: "Scale your Product",
      icon: <TrendingUp className="w-[60%] h-[60%]" />,
      iconPosition: "between",
    },
    {
      text: "with TS-UI",
    },
  ]}
/>
```

## Props

### IconicHeading

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| `lines` | `Line[]` | `—` | Array of lines to render |
| `align` | `"center" \| "left" \| "right"` | `"center"` | Text alignment |
| `className` | `string` | `—` | Additional classes for the heading |
| `shadow` | `boolean` | `false` | Adds a drop shadow to the heading |

### Line

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| `text` | `string` | `—` | Text content of the line |
| `icon` | `ReactNode` | `—` | Icon to render alongside the text |
| `iconPosition` | `"start" \| "end" \| "between"` | `—` | Where to place the icon relative to text |
| `iconClassName` | `string` | `—` | Additional classes for the icon wrapper |
| `muted` | `boolean` | `—` | Renders the line in muted foreground color |
| `className` | `string` | `—` | Additional classes for the line |
