import React, { useState, useEffect } from "react";

import { GlassDock } from "@/registry/glass-dock";

  const { theme, setTheme } = useTheme();
  const [mounted, setMounted] = useState(false);

  useEffect(() => {
    setMounted(true);
  }, []);

  const dockItems = [
    { 
      label: "Home", 
      icon: Home, 
      onClick: () => window.scrollTo({ top: 0, behavior: 'smooth' }) 
    },
    { 
      label: "Registry",

    },
    {

    },
    {
      label: "Theme",
      icon: theme === "dark" ? Sun : Moon,
      onClick: () => setTheme(theme === "dark" ? "light" : "dark"),
    },
  ];

  if (!mounted) return null;

  return (
    <div className="relative mt-10 flex justify-center  pointer-events-none">
        <div className="pointer-events-auto z-50">
          <GlassDock items={dockItems} />
        </div>
      </div>
  );

## Installation

```bash
npx shadcn@latest add https://senko-ui.vercel.app/r/glass-dock.json
```

## Usage

```tsx
import {GlassDock} from "@/registry/glass-dock"
```

```tsx
<GlassDock items={dockItems} />
```

## Props

| Prop         | Type        | Default | Description                                               |
| ------------ | ----------- | ------- | --------------------------------------------------------- |
| `swipeWidth` | `number`    | `100`   | Width of the sliding fill effect on hover as a percentage |
| `className`  | `string`    | `—`     | Additional classes to apply                               |
| `children`   | `ReactNode` | `—`     | Content inside the badge                                  |
